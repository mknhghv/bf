-- 1. 自動武裝色 (Buso)
local function callBuso()
    local char = player.Character
    if char and not char:FindFirstChild("HasBuso") then
        local remotes = ReplicatedStorage:FindFirstChild("Remotes")
        local commF = remotes and remotes:FindFirstChild("CommF_")
        if commF then 
            pcall(function() commF:InvokeServer("Buso") end) 
        end
    end
end

-- 2. 自動 V3
local function callV3()
    local remotes = ReplicatedStorage:FindFirstChild("Remotes")
    local commE = remotes and remotes:FindFirstChild("CommE")
    if commE then 
        pcall(function() commE:FireServer("ActivateAbility") end) 
    end
end

-- 3. 自動 V4
local function callV4()
    local backpack = player:FindFirstChild("Backpack")
    local awakening = backpack and backpack:FindFirstChild("Awakening")
    local remote = awakening and awakening:FindFirstChild("RemoteFunction")
    if remote then 
        pcall(function() remote:InvokeServer(true) end) 
    end
end

-- 4. 任務處理器
local function toggleTask(name, enabled, func)
    State[name] = enabled
    if State.Tasks[name] then 
        task.cancel(State.Tasks[name]) 
        State.Tasks[name] = nil
    end
    if enabled then
        State.Tasks[name] = task.spawn(function()
            while State[name] do
                func()
                task.wait(1.5)
            end
        end)
    end
end

-- 地牢出口傳送邏輯 (完整保留原始檔案功能)
local function getTPs()
    local tbl = {}
    local map = workspace:FindFirstChild("Map")
    local d = map and map:FindFirstChild("Dungeon")
    if not d then return tbl end
    for _, l in pairs(d:GetChildren()) do
        local n = tonumber(l.Name)
        local tp = l:FindFirstChild(CONFIG.TargetTeleporterName)
        local root = tp and (tp:FindFirstChild("Root") or tp:FindFirstChildWhichIsA("BasePart"))
        if n and root then table.insert(tbl, {level = n, root = root}) end
    end
    table.sort(tbl, function(a,b) return a.level < b.level end)
    return tbl
end

local function doTP(part, lv)
    if State.isTeleporting then return end
    State.isTeleporting = true
    StatusLabel.Text = "狀態：傳送至 Lv." .. lv
    if CONFIG.DisableMovement then humanoid.WalkSpeed = 0 end
    local tw = TweenService:Create(hrp, TweenInfo.new(CONFIG.TweenTime), {CFrame = part.CFrame})
    tw:Play()
    tw.Completed:Connect(function()
        if CONFIG.DisableMovement then humanoid.WalkSpeed = 16 end
        task.wait(0.4)
        State.isTeleporting = false
        StatusLabel.Text = "狀態：待機中"
    end)
end

ManualBtn.MouseButton1Click:Connect(function()
    local tps = getTPs()
    if #tps > 0 then doTP(tps[#tps].root, tps[#tps].level) end
end)

task.spawn(function()
    while true do
        task.wait(CONFIG.CheckInterval)
        if State.isAutoDetectEnabled and not State.isTeleporting then
            local tps = getTPs()
            for _, info in ipairs(tps) do
                if (hrp.Position - info.root.Position).Magnitude < CONFIG.DetectionRadius then
                    doTP(info.root, info.level)
                    break
                end
            end
        end
    end
end)
