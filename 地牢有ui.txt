local TweenService = game:GetService("TweenService")
local Players = game:GetService("Players")
local UserInputService = game:GetService("UserInputService")
local ReplicatedStorage = game:GetService("ReplicatedStorage")

local player = Players.LocalPlayer
local character, hrp, humanoid

-- ━━━━━━━━━━━━━━━━━━━━━━ 核心設定 ━━━━━━━━━━━━━━━━━━━━━━
local CONFIG = {
    CheckInterval = 0.8,
    TweenTime = 0.8,
    DetectionRadius = 2500,
}

local State = {
    isTeleporting = false,
    isAutoDetect = true,
    AutoBuso = false,
    AutoV3 = false,
    AutoV4 = false, -- 使用你原本的遠端觸發邏輯
    LockedLevel = 1
}

local function getRef()
    character = player.Character
    if character then
        hrp = character:FindFirstChild("HumanoidRootPart")
        humanoid = character:FindFirstChild("Humanoid")
    end
end

-- [你原本給的 V4 邏輯]
local function callV4()
    if State.AutoV4 then
        local b = player:FindFirstChild("Backpack")
        local a = b and b:FindFirstChild("Awakening")
        local r = a and a:FindFirstChild("RemoteFunction")
        if r then 
            pcall(function() 
                r:InvokeServer(true) 
            end) 
        end
    end
end

-- [自動補 Buff]
local function doBuffs()
    if not character then return end
    -- 武裝色
    if State.AutoBuso and not character:FindFirstChild("HasBuso") then
        pcall(function() ReplicatedStorage.Remotes.CommF_:InvokeServer("Buso") end)
    end
    -- V3 技能
    if State.AutoV3 then
        pcall(function() ReplicatedStorage.Remotes.CommE:FireServer("ActivateAbility") end)
    end
end

-- ━━━━━━━━━━━━━━━━━━━━━━ UI 建立 ━━━━━━━━━━━━━━━━━━━━━━
local sg = Instance.new("ScreenGui", player:WaitForChild("PlayerGui"))
sg.Name = "OriginalV4CoreUI"
sg.ResetOnSpawn = false

local main = Instance.new("Frame", sg)
main.Size = UDim2.new(0, 260, 0, 420)
main.Position = UDim2.new(0.5, -130, 0.4, -210)
main.BackgroundColor3 = Color3.new(0, 0, 0)
main.Active = true
main.Draggable = true
Instance.new("UICorner", main)

local bar = Instance.new("TextLabel", main)
bar.Size = UDim2.new(1, 0, 0, 50)
bar.BackgroundColor3 = Color3.fromRGB(40, 40, 40)
bar.Text = "自動地下城"
bar.TextColor3 = Color3.new(1, 1, 1)
bar.TextSize = 18
bar.Font = Enum.Font.GothamBold
bar.TextXAlignment = Enum.TextXAlignment.Left
Instance.new("UICorner", bar)

local close = Instance.new("TextButton", main)
close.Size = UDim2.new(0, 35, 0, 35)
close.Position = UDim2.new(1, -40, 0, 7)
close.Text = "X"
close.BackgroundColor3 = Color3.fromRGB(200, 50, 50)
close.TextColor3 = Color3.new(1, 1, 1)
Instance.new("UICorner", close)
close.MouseButton1Click:Connect(function() sg:Destroy() end)

local function makeBtn(txt, y, color, cb)
    local b = Instance.new("TextButton", main)
    b.Size = UDim2.new(0.9, 0, 0, 45)
    b.Position = UDim2.new(0.05, 0, 0, y)
    b.BackgroundColor3 = color
    b.Text = txt
    b.TextSize = 16
    b.TextColor3 = Color3.new(1, 1, 1)
    b.Font = Enum.Font.GothamSemibold
    Instance.new("UICorner", b)
    b.MouseButton1Click:Connect(function() cb(b) end)
    return b
end

-- 按鈕排版
makeBtn("自動 V4 技能", 65, Color3.fromRGB(60, 60, 70), function(b)
    State.AutoV4 = not State.AutoV4
    b.BackgroundColor3 = State.AutoV4 and Color3.fromRGB(0, 150, 80) or Color3.fromRGB(60, 60, 70)
end)

makeBtn("自動武裝色 (Buso)", 120, Color3.fromRGB(60, 60, 70), function(b)
    State.AutoBuso = not State.AutoBuso
    b.BackgroundColor3 = State.AutoBuso and Color3.fromRGB(0, 150, 80) or Color3.fromRGB(60, 60, 70)
end)

makeBtn("自動 V3 技能", 175, Color3.fromRGB(60, 60, 70), function(b)
    State.AutoV3 = not State.AutoV3
    b.BackgroundColor3 = State.AutoV3 and Color3.fromRGB(0, 150, 80) or Color3.fromRGB(60, 60, 70)
end)

makeBtn("自動偵測傳送：開啟", 235, Color3.fromRGB(0, 120, 60), function(b)
    State.isAutoDetect = not State.isAutoDetect
    b.Text = "自動偵測傳送：" .. (State.isAutoDetect and "開啟" or "關閉")
    b.BackgroundColor3 = State.isAutoDetect and Color3.fromRGB(0, 120, 60) or Color3.fromRGB(150, 0, 0)
end)

makeBtn("📍 強制傳送目前出口 📍", 295, Color3.fromRGB(100, 80, 200), function()
    State.isTeleporting = false
    getRef()
    local d = workspace:FindFirstChild("Map") and workspace.Map:FindFirstChild("Dungeon")
    if d then
        local f = d:FindFirstChild(tostring(State.LockedLevel))
        local exit = f and (f:FindFirstChild("ExitTeleporter") or f:FindFirstChild("Teleport"))
        local r = exit and (exit:FindFirstChild("Root") or exit:FindFirstChildWhichIsA("BasePart"))
        if r then doTP(r) end
    end
end)

local status = Instance.new("TextLabel", main)
status.Size = UDim2.new(1, 0, 0, 60)
status.Position = UDim2.new(0, 0, 1, -65)
status.Text = "目前關卡：第 1 關\n狀態：待機中"
status.TextSize = 18
status.TextColor3 = Color3.new(1, 1, 0)
status.BackgroundTransparency = 1
status.Parent = main

-- ━━━━━━━━━━━━━━━━━━━━━━ 核心循環 ━━━━━━━━━━━━━━━━━━━━━━

function doTP(part)
    if State.isTeleporting then return end
    getRef()
    State.isTeleporting = true
    local tw = TweenService:Create(hrp, TweenInfo.new(CONFIG.TweenTime), {CFrame = part.CFrame * CFrame.new(0, 5, 0)})
    tw:Play()
    tw.Completed:Connect(function() task.wait(0.5) State.isTeleporting = false end)
end

task.spawn(function()
    while true do
        task.wait(0.8)
        getRef()
        if not character or not hrp then continue end

        -- 執行 Buffs 與 原版 V4 邏輯
        doBuffs()
        callV4()

        -- 循序地牢邏輯
        local map = workspace:FindFirstChild("Map")
        local dungeon = map and map:FindFirstChild("Dungeon")
        if dungeon then
            for _, lv in pairs(dungeon:GetChildren()) do
                local n = tonumber(lv.Name)
                local floor = lv:FindFirstChild("Floor") or lv:FindFirstChildWhichIsA("BasePart")
                if n and floor and (hrp.Position - floor.Position).Magnitude < 600 then
                    State.LockedLevel = n
                end
            end

            if State.isAutoDetect and not State.isTeleporting then
                local folder = dungeon:FindFirstChild(tostring(State.LockedLevel))
                local exit = folder and (folder:FindFirstChild("ExitTeleporter") or folder:FindFirstChild("Teleport"))
                local root = exit and (exit:FindFirstChild("Root") or exit:FindFirstChildWhichIsA("BasePart"))
                if root and (hrp.Position - root.Position).Magnitude < CONFIG.DetectionRadius then
                    doTP(root)
                end
            end
        end
        status.Text = "目前關卡：第 " .. tostring(State.LockedLevel) .. " 關\n狀態：" .. (State.isTeleporting and "傳送中" or "待機中")
    end
end)
